Unir dos vectores:

      #Crear dos arreglos de 5 posiciones, A con valore de 0 a 99 y B de 99 a 200 luego unir esos arreglos.
      A = np.linspace(0,99,5)
      B = np.linspace(99,200,5)
      C = np.concatenate((A,B))

      # Unir verticalmente 2 vectores:
      _seed = 5
      npRdn = np.random.default_rng(_seed)
      A = npRdn.integers(100, size=(10,4))
      B = npRdn.integers(100, size=(10,4))

      C = np.vstack((A, B))

      # Unir horizontalmente
      C = np.hstack((A, B))

Sumar dos vectores:

      #Crear dos arreglos de 5 posiciones, A con valore de 0 a 99 y B de 99 a 200, esos arreglos deben de sumarse
      A = np.linspace(0,99,5)
      B = np.linspace(99,200,5)
      C = A + B
